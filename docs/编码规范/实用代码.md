// yield 
var fetch = require("node-fetch");

function* gen() { // 这里的 * 可以看成 async
  var url = "https://api.github.com/users/github";
  var result = yield fetch(url); // 这里的 yield 可以看成 await
  console.log(result);
}

var g = gen();
var result = g.next();
result.value.then(data => data.json()).then(data => g.next(data));

// async await
async function getData(url) {
  const res = await fetch(url)
  }
getData('https://api.github.com/users/github')

#### V-Model 实现
```java
<ProductReturnMaterialItemForm ref="itemFormRef" v-model="formData.produceReturnItems" />

const props = defineProps<{
  modelValue: Array<{
    productId: string
    outCount: number
  }>
  disabled: boolean
}>()
    
const emit = defineEmits(['update:modelValue'])

watch(
  () => props.modelValue,
  (val) => {
    formData.value = val.map((i) => ({
      ...i,
      outCount: i.outCount || 1
    }))
  },
  { immediate: true }
)

watch(
  formData,
  (val) => {
    if (JSON.stringify(val) !== JSON.stringify(props.modelValue)) {
      emit('update:modelValue', val)
    }
  },
  { deep: true }
)
```

```html
<!DOCTYPE html>

  </head>

    <div id="app">
      <h1>
        {{message}}
      </h1>
      <button @click="sayHello">阿牛</button>
    </div>
    <script type="module">
      // 单一模组写法
      import {createApp} from 'https://cdn.bootcdn.net/ajax/libs/vue/3.2.0-beta.7/vue.esm-browser.min.js'
      createApp({
        data: function () {
          return {
            message: "lalal"
          }
        },
        // 物件
        methods: {
          sayHello() {
            console.log(this.message)
          }
        },
        // 函式
        mounted() {
          console.log(this)
        }
      }).mount("#app")
    </script>
  </body>
</html>
```

### Gsap动画库示例
```html
<!DOCTYPE html>
<html lang="en">
  
  <head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Vue2中结合Gsap动画库示例代码</title>
    <script src="https://cdn.jsdelivr.net/npm/vue@2.6.14/dist/vue.js"></script>
    <script src="https://cdnjs.cloudflare.com/ajax/libs/gsap/3.2.4/gsap.min.js"></script>
    
  </head>
  
  <body>
    <div id="animated-number-demo">
      <input v-model.number="number" type="number" step="20">
      <p>{{ animatedNumber }}</p >
    </div>
    <script>
      let data = {
        number: 0,
        tweenedNumber: 0
      }
      
      let app = new Vue({
        el: '#animated-number-demo',
        data: data,
        computed: {
          animatedNumber: function () {
            return this.tweenedNumber.toFixed(0);
          }
        },
        watch: {
          number: function (newValue) {
            gsap.to(this.$data, {
              duration: 0.5,
              tweenedNumber: newValue
            });
          }
        }
      })
    </script>
  </body>
  
</html>
```
